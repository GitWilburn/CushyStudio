# Practical Shell Aliases

```bash
alias supabase='./node_modules/.bin/supabase'

# use your own OS / arch
alias npm='.cushy/node/v18.19.0-darwin-arm64/bin/npm'
alias node='.cushy/node/v18.19.0-darwin-arm64/bin/node'
```
